### P1-1: 引入 Vue Router

**目标**: 替换手动 v-if/v-else 组件切换，实现标准客户端路由

#### 1. 安装依赖
```bash
npm install vue-router@4
```

#### 2. 创建路由配置
**文件**: `src/router/index.js`

```javascript
import { createRouter, createWebHistory } from 'vue-router'

const routes = [
  { path: '/', name: 'landing', component: () => import('@/components/ArchWorkspaceNew.vue'), meta: { title: '工作台' } },
  { path: '/diagram', name: 'diagram', component: () => import('@/views/AADiagramApp/index.vue'), meta: { title: '架构图' } },
  { path: '/config', name: 'config', component: () => import('@/components/ConfigApp.vue'), meta: { title: '配置' } },
  { path: '/data/:productId?/:versionId?', name: 'data', component: () => import('@/views/ArchDataManageApp/index.vue'), meta: { title: '架构数据管理' } },
  { path: '/product-version', name: 'product-version', component: () => import('@/views/ProductVersionApp/index.vue'), meta: { title: '产品版本' } },
  { path: '/system', name: 'system', component: () => import('@/views/SystemManagement/index.vue'), meta: { title: '系统管理' } },
  { path: '/test', name: 'test', component: () => import('@/views/ComponentTest.vue'), meta: { title: '组件测试' } }
]

const router = createRouter({
  history: createWebHistory(),
  routes
})

router.beforeEach((to, from, next) => {
  document.title = to.meta.title ? `${to.meta.title} - ArchWorkspace` : 'ArchWorkspace'
  next()
})

export default router
```

#### 3. 注册路由
**文件**: `src/main.js`

```javascript

```

#### 4. 重构 App.vue

## 1. 概述

### 1.1 目标

| 目标 | 具体内容 |
|------|---------|
| **架构迁移** | 替代旧的 TreeNavigator.vue，实现 4 层架构对象树形导航 |
| **通用能力** | 沉淀 CollapsiblePanel 为可复用折叠容器组件 |

### 1.2 交付物

| 组件 | 路径 | 职责 | 代码行数(估) |
|------|------|------|-------------|
| CollapsiblePanel.vue | `src/components/common/CollapsiblePanel/` | 通用折叠面板 | ~150 |
| ObjectTreePanel.vue | `src/components/business/ObjectTreePanel/` | 4层树形导航 | ~350 |

### 1.3 与现有组件的关系

| 现有组件 | 差异点 | 复用策略 |
|----------|--------|----------|
| `AppCollapse` | 卡片样式折叠，内容区样式不同 | 不直接复用，参考其折叠动画实现 |
| `MasterDetailLayout` | 主从布局，含 resizer | 侧边栏场景可组合使用，但不继承 |

## 2. CollapsiblePanel 组件设计

### 2.1 组件定位

**定位**: 通用折叠面板容器，支持标题、徽章、操作按钮、可调宽度。

**与 AppCollapse 的区别**:
- `AppCollapse`: 内容区内联展开，适合手风琴式交互
- `CollapsiblePanel`: 侧边栏容器，支持拖拽调宽，适合固定面板折叠

### 2.2 Props 定义

### 2.5 组件模板

```vue
<template>
  <div 
    :class="['collapsible-panel', { 'is-collapsed': !expanded }, props.class]"
    :style="containerStyle"
  >
    <div class="collapsible-panel__header" @click="handleToggle">
      <div class="collapsible-panel__header-left">
        <slot name="header">
          <span v-if="title" class="collapsible-panel__title">{{ title }}</span>
          <el-badge 
            v-if="badge" 
            :value="badge" 
            class="collapsible-panel__badge"
          />
        </slot>
      </div>
      
      <div class="collapsible-panel__header-right">
        <slot name="extra" />
        <button 
          v-if="collapsible" 
          class="collapsible-panel__toggle"
          type="button"
        >
          <el-icon>
            <ArrowRight v-if="!expanded" />
            <ArrowDown v-else />
          </el-icon>
        </button>
      </div>
    </div>
    
    <div v-show="expanded" class="collapsible-panel__content">
      <slot />
    </div>
    
    <div 
      v-if="resizable && expanded"
      class="collapsible-panel__resizer"
      @mousedown="startResize"
    />
  </div>
</template>

<script setup>
import { ref, computed, watch } from 'vue'
import { ArrowRight, ArrowDown } from '@element-plus/icons-vue'

const props = withDefaults(defineProps<CollapsiblePanelProps>(), {
  title: '',
  badge: null,
  collapsible: true,
  defaultExpanded: true,
  resizable: false,
  minWidth: 200,
  maxWidth: 600,
  defaultWidth: 280,
  collapsePosition: 'header',
  heightFull: true
})

const emit = defineEmits([
  'update:expanded',
  'update:width',
  'toggle',
  'resize'
])

const expanded = ref(props.defaultExpanded)
const currentWidth = ref(props.defaultWidth)
const isResizing = ref(false)

const containerStyle = computed(() => ({
  ...(props.heightFull ? { height: '100%' } : {}),
  ...(!expanded.value ? { width: '0px', minWidth: '0px' } : {}),
  ...(expanded.value ? { width: `${currentWidth.value}px` } : {})
}))

watch(() => props.defaultExpanded, (val) => {
  expanded.value = val
})

function handleToggle() {
  if (!props.collapsible) return
  expanded.value = !expanded.value
  emit('update:expanded', expanded.value)
  emit('toggle', expanded.value)
}

function startResize(e) {
  if (!props.resizable || !expanded.value) return
  
  isResizing.value = true
  const startX = e.clientX
  const startWidth = currentWidth.value
  
  document.addEventListener('mousemove', handleResize)
  document.addEventListener('mouseup', stopResize)
  
  function handleResize(e) {
    const delta = e.clientX - startX
    const newWidth = Math.min(props.maxWidth, Math.max(props.minWidth, startWidth + delta))
    currentWidth.value = newWidth
  }
  
  function stopResize() {
    isResizing.value = false
    document.removeEventListener('mousemove', handleResize)
    document.removeEventListener('mouseup', stopResize)
    emit('update:width', currentWidth.value)
    emit('resize', currentWidth.value)
  }
}
</script>
```

## 3. ObjectTreePanel 组件设计

### 3.1 组件定位

**定位**: 4 层架构对象树形导航，基于 el-tree 封装。

**元数据驱动**: 树层级结构、显示字段从 YAML `hierarchies` 配置读取，不硬编码对象类型。

### 3.2 Props 定义

```typescript
interface ObjectTreePanelProps {
  /** 根节点对象类型，默认从 hierarchies.root_type 读取 */
  objectType?: string
  /** 版本上下文 ID (必填) */
  versionId: number
  /** 已选中的节点 ID 列表 */
  selectedNodes?: number[]
  /** 是否显示复选框 */
  showCheckbox?: boolean
  /** 是否显示子对象计数 */
  showCount?: boolean
  /** 搜索框占位符 */
  searchPlaceholder?: string
  /** 默认展开全部 */
  defaultExpandAll?: boolean
  /** 是否懒加载子节点 */
  lazyLoad?: boolean
  /** 懒加载时的子节点深度 */
  lazyLoadDepth?: number
  /** 节点过滤函数 */
  filterMethod?: (value: string, data: TreeNode) => boolean
  /** 自定义节点渲染 */
  nodeRenderer?: (h: h, context: NodeRenderContext) => VNode
}
```

### 3.3 Events 定义

```typescript
interface ObjectTreePanelEvents {
  /** 节点点击选中 */
  'select': (data: TreeNode, node: ElTreeNode) => void
  /** 节点勾选变化 */
  'check': (data: TreeNode, checked: { checkedKeys: number[], checkedNodes: TreeNode[] }) => void
  /** 树加载完成 */
  'load': (treeData: TreeNode[]) => void
  /** 节点展开 */
  'expand': (data: TreeNode, node: ElTreeNode, resolved: boolean) => void
  /** 节点折叠 */
  'collapse': (data: TreeNode, node: ElTreeNode) => void
  /** 搜索变化 */
  'search': (query: string) => void
  /** 清空选择 */
  'clear': () => void
}
```

### 3.4 数据模型

```typescript
interface TreeNode {
  /** 节点 ID */
  id: number
  /** 节点名称 */
  name: string
  /** 节点类型: domain | sub_domain | service_module | business_object */
  type: string
  /** 父节点 ID */
  parentId?: number
  /** 子节点列表 */
  children?: TreeNode[]
  /** 子对象计数 */
  childrenCount?: number
  /** 是否禁用 (无子对象时) */
  disabled?: boolean
  /** 是否叶子节点 */
  isLeaf?: boolean
  /** 层级深度 */
  level?: number
  /** 完整路径 (用于面包屑) */
  path?: HierarchyPath[]
  /** 额外属性 */
  [key: string]: any
}

interface HierarchyPath {
  type: string
  id: number
  name: string
}
```

### 3.5 元数据驱动设计

```typescript
// ObjectTreePanel.vue - 核心元数据驱动逻辑
const props = defineProps<ObjectTreePanelProps>()

// 从 provide 获取元数据配置
const metaObject = inject('metaObject', ref(null))

// 层级配置 - 从 YAML hierarchies 读取
const hierarchyConfig = computed(() => {
  return metaObject.value?.hierarchies?.[0] || {
    levels: [
      { object_type: 'domain', children_field: 'sub_domains' },
      { object_type: 'sub_domain', children_field: 'service_modules' },
      { object_type: 'service_module', children_field: 'business_objects' },
      { object_type: 'business_object', children_field: null }
    ]
  }
})

// 根节点类型 - 动态获取，不硬编码
const rootType = computed(() => hierarchyConfig.value.root_type || 'domain')

// 子节点类型映射 - 从 YAML levels 读取
const childFieldMap = computed(() => {
  const map = {}
  hierarchyConfig.value.levels?.forEach(level => {
    map[level.object_type] = level.children_field
  })
  return map
})

// 获取子节点类型 - 动态方法，不使用 if/else
function getChildObjectType(parentType: string): string | null {
  const index = hierarchyConfig.value.levels?.findIndex(
    l => l.object_type === parentType
  )
  if (index === undefined || index === -1) return null
  return hierarchyConfig.value.levels?.[index + 1]?.object_type || null
}

// 节点显示字段 - 从 YAML display_name.expression 读取
const nodeLabelField = computed(() => {
  return metaObject.value?.display_name?.expression?.split('.')?.[1] || 'name'
})
```

### 3.6 组件模板

```vue
<template>
  <div class="object-tree-panel">
    <!-- 搜索栏 -->
    <div v-if="showSearch" class="object-tree-panel__search">
      <el-input
        v-model="searchQuery"
        :placeholder="searchPlaceholder"
        clearable
        @input="handleSearch"
      >
        <template #prefix>
          <el-icon><Search /></el-icon>
        </template>
      </el-input>
    </div>
    
    <!-- 工具栏 -->
    <div v-if="showToolbar" class="object-tree-panel__toolbar">
      <el-button size="small" link @click="handleExpandAll">展开</el-button>
      <el-button size="small" link @click="handleCollapseAll">折叠</el-button>
      <el-button 
        v-if="showCheckbox" 
        size="small" 
        link 
        @click="handleSelectAll"
      >
        全选
      </el-button>
      <el-button 
        v-if="showCheckbox" 
        size="small" 
        link 
        @click="handleClearAll"
      >
        清空

    </div>
    
    <!-- 树容器 -->
    <div class="object-tree-panel__tree">
      <el-tree
        ref="treeRef"
        :data="treeData"
        :props="treeProps"
        :node-key="nodeKey"
        :default-expand-all="defaultExpandAll"
        :show-checkbox="showCheckbox"
        :expand-on-click-node="false"
        :filter-node-method="filterNodeMethod"
        :lazy="lazyLoad"
        :load="handleLazyLoad"
        highlight-current
        @node-click="handleNodeClick"
        @check="handleCheck"
        @node-expand="handleNodeExpand"
        @node-collapse="handleNodeCollapse"
      >
        <template #default="{ node, data }">
          <span class="object-tree-panel__node">
            <span class="object-tree-panel__node-label">
              {{ getNodeLabel(data) }}
            </span>
            <span 
              v-if="showCount && data.childrenCount > 0" 
              class="object-tree-panel__node-count"
            >
              ({{ data.childrenCount }})
            </span>
          </span>
        </template>
      </el-tree>
      
      <!-- 空状态 -->
      <div v-if="loading" class="object-tree-panel__loading">
        <el-icon class="is-loading"><Loading /></el-icon>
        <span>加载中...</span>
      </div>
      
      <div v-else-if="!treeData.length" class="object-tree-panel__empty">
        <el-empty description="暂无数据" />
      </div>
    </div>
  </div>
</template>

<script setup>
import { ref, computed, watch, onMounted } from 'vue'
import { Search, Loading } from '@element-plus/icons-vue'
import { boService } from '@/services/boService'

const props = withDefaults(defineProps<ObjectTreePanelProps>(), {
  objectType: '',
  selectedNodes: () => [],
  showCheckbox: true,
  showCount: true,
  searchPlaceholder: '搜索对象',
  defaultExpandAll: false,
  lazyLoad: false,
  lazyLoadDepth: 2
})

const emit = defineEmits<ObjectTreePanelEvents>()

// 状态
const treeRef = ref(null)
const treeData = ref([])
const loading = ref(false)
const searchQuery = ref('')
const expandedKeys = ref([])

// 元数据配置
const metaObject = inject('metaObject', ref(null))

const hierarchyConfig = computed(() => {
  return metaObject.value?.hierarchies?.[0] || getDefaultHierarchyConfig()
})

function getDefaultHierarchyConfig() {
  return {
    levels: [
      { object_type: 'domain', children_field: 'sub_domains' },
      { object_type: 'sub_domain', children_field: 'service_modules' },
      { object_type: 'service_module', children_field: 'business_objects' },
      { object_type: 'business_object', children_field: null }
    ],
    root_type: 'domain',
    root_filter: 'version_id'
  }
}

const rootType = computed(() => hierarchyConfig.value.root_type || 'domain')
const rootFilterField = computed(() => hierarchyConfig.value.root_filter || 'version_id')

const treeProps = computed(() => ({
  label: nodeLabelField.value,
  children: childFieldMap.value[props.objectType || rootType.value] || 'children',
  isLeaf: (data) => {
    const childType = getChildObjectType(data.type)
    return !childType
  }
}))

const nodeKey = 'id'
const showSearch = computed(() => true)
const showToolbar = computed(() => true)

const nodeLabelField = computed(() => {
  return metaObject.value?.display_name?.expression?.split('.')?.[1] || 'name'
})

const childFieldMap = computed(() => {
  const map = {}
  hierarchyConfig.value.levels?.forEach(level => {
    map[level.object_type] = level.children_field
  })
  return map
})

function getChildObjectType(parentType) {
  const levels = hierarchyConfig.value.levels || []
  const index = levels.findIndex(l => l.object_type === parentType)
  if (index === -1 || index >= levels.length - 1) return null
  return levels[index + 1].object_type
}

function getNodeLabel(data) {
  return data[nodeLabelField.value] || data.name || data.label || ''
}

// 加载树数据
async function loadTreeData() {
  if (!props.versionId) return
  
  loading.value = true
  try {
    const params = {
      [rootFilterField.value]: props.versionId,
      include_counts: props.showCount
    }
    
    const result = await boService.query(rootType.value, {
      filters: params,
      pageSize: 1000
    })
    
    if (result.success) {
      treeData.value = normalizeTreeData(result.data?.items || result.data || [])
      emit('load', treeData.value)
    }
  } catch (error) {
    console.error('[ObjectTreePanel] Failed to load tree:', error)
  } finally {
    loading.value = false
  }
}

// 规范化树数据
function normalizeTreeData(items) {
  return items.map(item => ({
    id: item.id,
    name: item.name,
    type: props.objectType || rootType.value,
    parentId: item.parent_id,
    childrenCount: item.children_count || 0,
    disabled: !item.children_count && !hasChildren(item),
    isLeaf: !hasChildren(item),
    ...item
  }))
}

function hasChildren(item) {
  const childField = childFieldMap.value[item.type]
  return childField && item[childField]?.length > 0
}

// 懒加载子节点
async function handleLazyLoad(node, resolve) {
  if (node.level === 0) {
    return resolve([])
  }
  
  const childType = getChildObjectType(node.data.type)
  if (!childType) {
    return resolve([])
  }
  
  try {
    const result = await boService.query(childType, {
      filters: { [node.data.type + '_id']: node.data.id }
    })
    
    if (result.success) {
      const children = normalizeTreeData(result.data?.items || result.data || [])
      resolve(children)
    } else {
      resolve([])
    }
  } catch {
    resolve([])
  }
}

// 节点点击
function handleNodeClick(data, node) {
  emit('select', data, node)
}

// 节点勾选
function handleCheck(data, { checkedKeys, checkedNodes }) {
  emit('check', data, { checkedKeys, checkedNodes })
}

// 节点展开
function handleNodeExpand(data, node) {
  expandedKeys.value.push(data.id)
  emit('expand', data, node, true)
}

// 节点折叠
function handleNodeCollapse(data, node) {
  const index = expandedKeys.value.indexOf(data.id)
  if (index > -1) expandedKeys.value.splice(index, 1)
  emit('collapse', data, node)
}

// 搜索过滤
function filterNodeMethod(value, data) {
  if (!value) return true
  return getNodeLabel(data).toLowerCase().includes(value.toLowerCase())
}

function handleSearch(query) {
  treeRef.value?.filter(query)
  emit('search', query)
}

// 全选
function handleSelectAll() {
  const keys = treeData.value.map(item => item.id)
  treeRef.value?.setCheckedKeys(keys)
}

// 清空
function handleClearAll() {
  treeRef.value?.setCheckedKeys([])
  emit('clear')
}

// 展开全部
function handleExpandAll() {
  treeData.value.forEach(data => {
    treeRef.value?.expandNode(data, true)
  })
}

// 折叠全部
function handleCollapseAll() {
  treeRef.value?.collapseAll()
}

// 获取选中的节点
function getCheckedNodes() {
  return treeRef.value?.getCheckedNodes() || []
}

function getCheckedKeys() {
  return treeRef.value?.getCheckedKeys() || []
}

function getCurrentNode() {
  return treeRef.value?.getCurrentNode()
}

function getNodeById(id) {
  return treeRef.value?.getNode(id)
}

// 暴露方法
defineExpose({
  loadTreeData,
  getCheckedNodes,
  getCheckedKeys,
  getCurrentNode,
  getNodeById,
  expandAll: handleExpandAll,
  collapseAll: handleCollapseAll
})

// 监听版本变化
watch(() => props.versionId, () => {
  loadTreeData()
}, { immediate: true })

// 监听选中节点变化
watch(() => props.selectedNodes, (newVal) => {
  if (treeRef.value && newVal?.length) {
    treeRef.value.setCheckedKeys(newVal)
  }
})
</script>
```

**接口**: `GET /api/v2/meta/hierarchy/tree`

**请求参数**:

```typescript
interface HierarchyTreeRequest {
  version_id: number       // [必填] 版本 ID

  max_depth?: number       // [可选] 最大展开深度，默认 4
```

**响应格式**:

```typescript
interface HierarchyTreeResponse {
```

### 4.3 boService 扩展

在 `src/services/boService.js` 中添加层级相关方法：

```javascript
// 在 BOService 类中添加

/**
 * 获取层级树数据
 * @param {string} rootType - 根节点类型
 * @param {Object} params - 查询参数
 * @returns {Promise<Object>} 树数据
 */
async getHierarchyTree(rootType, params = {}) {
  const queryStr = new URLSearchParams(
    Object.entries(params)
      .filter(([_, v]) => v !== undefined && v !== null && v !== '')
      .map(([k, v]) => [k, String(v)])
  ).toString()

  const url = `${this.API_BASE_V2}/meta/hierarchy/tree${queryStr ? '?' + queryStr : ''}`

  const response = await fetch(url, {
    method: 'GET',
    headers: this._getHeaders()
  })

  return this._handleResponse(response)
}

/**
 * 获取子节点计数
 * @param {string} objectType - 对象类型
 * @param {number} id - 对象 ID
 * @param {Object} params - 查询参数
 * @returns {Promise<Object>} 计数结果
 */
async getChildCount(objectType, id, params = {}) {
  const queryStr = new URLSearchParams(
    Object.entries(params)
      .filter(([_, v]) => v !== undefined && v !== null && v !== '')
      .map(([k, v]) => [k, String(v)])
  ).toString()

  const url = `${this.API_BASE_V2}/bo/${objectType}/${id}/child-count${queryStr ? '?' + queryStr : ''}`

  const response = await fetch(url, {
    method: 'GET',
    headers: this._getHeaders()
  })

  return this._handleResponse(response)
}
```

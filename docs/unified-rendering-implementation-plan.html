分组模型统一渲染 — 安全分步执行方案

6个Phase · 18个Step · 每步可验证 · 每步可回滚

安全原则

三条铁律：

  • 增量式变更：每一步只做一件事，新增代码不删除旧代码，新旧并行运行
  • 对比验证：每一步完成后，新旧路径输出必须一致，否则不进入下一步
  • 即时回滚：每一步都有明确的回滚方法，回滚只需删除新增代码、恢复旧代码

1.2 新建enrichGroupModel函数 新增

文件：groupModel/enrichGroupModel.js（新文件）

操作：创建enrichGroupModel函数，为GroupModel树补充颜色和标注信息：

export function enrichGroupModel(groupModel, chartType, options) {
  const { colorMap, centerCodes, annotationMap, nodeTextColor } = options
  
  function enrichGroup(groupId) {
    const group = groupModel.getById(groupId)
    if (!group) return
    
    const isTerminal = isTerminalGroup(group, chartType)
    
    if (isTerminal) {
      const code = group.elementRef?.code
      if (code) {
        group.color = colorMap.get(code) || null
        group.textColor = nodeTextColor || 'black'
        group.isCenter = centerCodes.has(code)
        const annotation = annotationMap.get(code)
        if (annotation) {
          group.annotationCategory = annotation.category || 'info'
          group.annotationContent = annotation.content || ''
        }
      }
    } else {
      // 容器颜色：使用分组层级对应的颜色
      // domain级用domainColors，subDomain级用subDomainColors
      group.color = options.containerColorMap?.get(group.id) || null
    }
    
    // 递归处理子分组
    const children = groupModel.getChildren(groupId)
    children.forEach(child => enrichGroup(child.id))
  }
  
  // 从根分组开始
  groupModel.getRootGroups().forEach(root => enrichGroup(root.id))
}

安全性：新文件，不修改任何现有代码，不调用此函数则无影响

1.3 新建ColorCalculator 新增

文件：groupModel/ColorCalculator.js（新文件）

操作：创建统一的颜色计算器，输出 colorMap（code→color 映射）：

export class ColorCalculator {
  static compute(config) {
    const { nodes, colorGroupBy, colorScheme, centerScopeColor, customColors } = config
    const colors = COLOR_SCHEMES[colorScheme] || COLOR_SCHEMES.default
    const colorMap = new Map()
    
    // 1. 按colorGroupBy分组
    const groupKeys = new Map()
    nodes.forEach(node => {
      const key = colorGroupBy === 'subDomain' ? node.subDomain
                : colorGroupBy === 'serviceModule' ? node.serviceModule
                : node.domain
      groupKeys.set(node.code, key)
    })
    
    // 2. 为每个分组分配颜色
    const uniqueGroups = [...new Set(groupKeys.values())]
    const groupColorMap = new Map()
    uniqueGroups.forEach((group, index) => {
      groupColorMap.set(group, customColors[group] || colors[index % colors.length])
    })
    
    // 3. 逐节点计算最终颜色
    nodes.forEach(node => {
      const baseColor = groupColorMap.get(groupKeys.get(node.code)) || colors[0]
      const finalColor = node.isCenter ? centerScopeColor : baseColor
      colorMap.set(node.code, finalColor)
    })
    
    return { colorMap, groupColorMap }
  }
}

安全性：新文件，不修改任何现有代码

1.4 新建UnifiedRenderer 新增

文件：groupModel/UnifiedRenderer.js（新文件）

操作：创建统一渲染器，从enrichedGroupModel生成Mermaid代码：

export class UnifiedRenderer {
  static render(groupModel, links, chartType, options) {
    const rootGroups = groupModel.getRootGroups()
    const nodeIdMap = buildNodeIdMap(rootGroups, chartType)
    
    let code = options.layoutEngine === 'elk' 
      ? 'flowchart-elk LR\n' : 'flowchart LR\n'
    
    // 1. 递归渲染分组树
    rootGroups.forEach(group => {
      code += this.renderGroup(group, groupModel, chartType, nodeIdMap, 0)
    })
    
    // 2. 渲染连线
    links.forEach(link => {
      const sourceId = nodeIdMap.codeToIdMap.get(link.source)
      const targetId = nodeIdMap.codeToIdMap.get(link.target)
      if (sourceId && targetId) {
        code += `  ${sourceId} --> ${targetId}\n`
      }
    })
    
    // 3. 渲染样式
    rootGroups.forEach(group => {
      code += this.renderStyles(group, groupModel, chartType)
    })
    
    return code
  }
  
  static renderGroup(group, groupModel, chartType, nodeIdMap, depth) {
    const isTerminal = isTerminalGroup(group, chartType)
    let code = ''
    
    if (isTerminal) {
      const displayText = group.elementRef?.code 
        ? `${group.title}\\n(${group.elementRef.code})` 
        : group.title
      const centerMark = group.isCenter ? '◆ ' : ''
      code += `${'  '.repeat(depth + 1)}${group.id}["${centerMark}${displayText}"]\n`
    } else {
      code += `${'  '.repeat(depth + 1)}subgraph ${group.id}["${group.title}"]\n`
      code += `${'  '.repeat(depth + 2)}direction ${group.layout?.direction || 'TB'}\n`
      
      const children = groupModel.getChildren(group.id)
      children.forEach(child => {
        code += this.renderGroup(child, groupModel, chartType, nodeIdMap, depth + 1)
      })
      
      code += `${'  '.repeat(depth + 1)}end\n`
    }
    
    return code
  }
  
  static renderStyles(group, groupModel, chartType) {
    const isTerminal = isTerminalGroup(group, chartType)
    let code = ''
    
    if (isTerminal && group.color) {
      code += `  style ${group.id} fill:${group.color},color:${group.textColor || 'black'}\n`
    } else if (!isTerminal && group.color) {
      code += `  style ${group.id} fill:${group.color},stroke:#999\n`
    }
    
    const children = groupModel.getChildren(group.id)
    children.forEach(child => {
      code += this.renderStyles(child, groupModel, chartType)
    })
    
    return code
  }
}

安全性：新文件，不修改任何现有代码，不调用则无影响

总结

6个Phase，18个Step，每一步都可独立验证和回滚。

安全性保障的核心机制是feature flag：所有新代码只在useUnifiedRenderer=true时执行，默认为false。这意味着在任何时候，只需将flag设为false即可恢复旧渲染器，无需修改任何代码。

Phase 0-1是纯新增，零风险；Phase 2-3是验证阶段，通过对比确保新渲染器输出与旧渲染器一致；Phase 4是切换阶段，将默认值改为true；Phase 5是清理阶段，删除旧代码。每个Phase完成后才进入下一个Phase。

最大的风险点在Phase 3.3（BO图节点id从name改为code）。这是整个方案中唯一可能影响Mermaid渲染正确性的变更。建议在Phase 3.2中充分验证后再执行3.3。
